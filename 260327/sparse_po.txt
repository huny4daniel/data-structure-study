========================================================
sparse_po.c — 희소 다항식(Sparse Polynomial) 구현 정리
========================================================

이 파일은 0이 아닌 항만 저장하는 희소 다항식을 구조체로 표현하고,
덧셈과 특정 x값 대입 계산을 구현한 코드이다.


--------------------------------------------------------
[희소 다항식(Sparse Polynomial)이란?]
--------------------------------------------------------

다항식을 표현하는 방법에는 크게 두 가지가 있다.

1. 일반 배열 표현 (Dense Polynomial)
   - 인덱스 = 지수, 값 = 계수로 배열에 저장
   - 예: 10x^100 + 6  →  배열 크기 101개 필요, 나머지 99개는 모두 0
   - 차수가 높고 항이 적을수록 0이 대부분 → 메모리 낭비 심함

2. 희소 표현 (Sparse Polynomial)  ← 이 코드의 방식
   - 0이 아닌 항만 (지수, 계수) 쌍으로 저장
   - 예: 10x^100 + 6  →  {(100, 10.0), (0, 6.0)} 단 2개만 저장
   - 항이 적은 고차 다항식에서 메모리를 크게 절약할 수 있음

정리:
   희소 다항식 = "실제로 값이 있는 항만 골라서 저장하는 다항식 표현법"
   희소(Sparse)란 데이터가 드문드문 있다는 뜻으로,
   행렬에서도 같은 개념이 쓰인다 (0이 많은 행렬 = 희소 행렬).

[헤더 및 상수]
----------------------------------------------------------------

#include <stdio.h>
    - printf 사용을 위한 표준 입출력 헤더

#include <math.h>
    - pow() 함수 사용을 위한 수학 헤더

#define MAX_TERMS 100
    - 다항식이 가질 수 있는 최대 항의 수를 100으로 제한


----------------------------------------------------------------
[구조체 정의]
----------------------------------------------------------------

typedef struct {
    int exp;
    float coef;
} Term;

    - 하나의 항(term)을 표현하는 구조체
    - exp  : 지수 (예: x^3 에서 3)
    - coef : 계수 (예: 10x^3 에서 10)

typedef struct {
    int nTerms;
    Term term[MAX_TERMS];
} SparsePoly;

    - 다항식 전체를 표현하는 구조체
    - nTerms     : 실제 저장된 항의 개수
    - term[]     : 항들의 배열
    - 핵심: 0인 항은 저장하지 않아 메모리를 절약 → "희소(Sparse)" 표현


----------------------------------------------------------------
[evaluate 함수 — 다항식에 x값 대입]
----------------------------------------------------------------

float evaluate(SparsePoly a, float x) {
    float result = 0.f;
    - 결과값을 0으로 초기화

    for (int i = 0; i < a.nTerms; i++) {
        result += (float)(a.term[i].coef * pow(x, a.term[i].exp));
    }
    - 각 항을 순회하며 coef * x^exp 를 계산해서 누적 합산
    - 예: 10.0 * x^100 + 6.0 * x^0

    return result;
    - 최종 합산 결과 반환
}


----------------------------------------------------------------
[add 함수 — 두 다항식 덧셈]
----------------------------------------------------------------

두 다항식이 지수 내림차순으로 정렬되어 있다고 가정하고 병합(merge)한다.

SparsePoly add(SparsePoly a, SparsePoly b) {
    SparsePoly c;
    c.nTerms = 0;
    - 결과 다항식 c 초기화

    for (int i = 0, j = 0; i < a.nTerms || j < b.nTerms;) {
    - a의 인덱스 i, b의 인덱스 j를 동시에 사용
    - 둘 중 하나라도 항이 남아있으면 반복 계속

        if (i == a.nTerms || a.term[i].exp < b.term[j].exp) {
            c.term[c.nTerms++] = b.term[j++];
        }
        - a가 끝났거나, a의 지수 < b의 지수인 경우
        - 지수가 더 큰 b의 항을 c에 추가

        else if (j == b.nTerms || a.term[i].exp > b.term[j].exp) {
            c.term[c.nTerms++] = a.term[i++];
        }
        - b가 끝났거나, a의 지수 > b의 지수인 경우
        - 지수가 더 큰 a의 항을 c에 추가

        else {
            c.term[c.nTerms] = a.term[i++];
            c.term[c.nTerms++].coef += b.term[j++].coef;
        }
        - a와 b의 지수가 같은 경우 (동차항)
        - a의 항을 먼저 복사한 뒤, b의 계수를 더함 (동차항 합산)
    }

    return c;
    - 완성된 다항식 c 반환
}


----------------------------------------------------------------
[print_spoly 함수 — 다항식 출력]
----------------------------------------------------------------

void print_spoly(SparsePoly p, char str[]) {
    printf("%s = ", str);
    - 이름(str)을 먼저 출력 (예: "A = ")

    for (int i = 0; i < p.nTerms; i++) {
        printf("%5.1f", p.term[i].coef);
        - 계수를 소수 1자리, 최소 5칸 폭으로 출력

        if (p.term[i].exp > 0) {
            printf(" x^%d ", p.term[i].exp);
            - 지수가 0보다 크면 x^지수 형태로 출력
            - 지수가 0이면 상수항이므로 x를 붙이지 않음

            if (i < p.nTerms - 1) {
                printf("+ ");
            }
            - 마지막 항이 아니면 + 기호 출력
        }
    }
    printf("\n");
    - 전체 출력 후 줄바꿈
}


----------------------------------------------------------------
[main 함수]
----------------------------------------------------------------

SparsePoly a = {2, {{100, 10.0f}, {0, 6.0f}}};
    - a = 10x^100 + 6
    - 항 2개: 지수 100(계수 10.0), 지수 0(계수 6.0)

SparsePoly b = {2, {{50, 1.1f}, {3, 2.0f}}};
    - b = 1.1x^50 + 2x^3
    - 항 2개: 지수 50(계수 1.1), 지수 3(계수 2.0)

SparsePoly c = add(a, b);
    - a + b 계산
    - c = 10x^100 + 1.1x^50 + 2x^3 + 6

print_spoly(a, " A = ");
print_spoly(b, " B = ");
print_spoly(c, " A + B = ");
    - a, b, c 각각 출력

printf("A(1) = %f\n", evaluate(a, 1.0f));
    - A(1) = 10*1^100 + 6*1^0 = 10 + 6 = 16.0

printf("B(2) = %f\n", evaluate(b, 2.0f));
    - B(2) = 1.1*2^50 + 2*2^3 = 매우 큰 수 + 16
